### `frontend/src/api/videoExports.ts`

```typescript
export interface CreateVideoExportRequest {
  module: 'process' | 'console';
  graphic_id: string;
  range_start: number;  // epoch ms
  range_end: number;
  step_ms: number;
  fps: number;
  width_px: number;
  height_px: number;
  device_pixel_ratio: number;
  overlay_timestamp: boolean;
  crf: number;
}

export interface VideoExportJob {
  id: string;
  status: 'queued' | 'processing' | 'completed' | 'failed' | 'cancelled';
  module: string;
  graphic_id: string;
  range_start: string;
  range_end: string;
  step_ms: number;
  fps: number;
  width_px: number;
  height_px: number;
  overlay_timestamp: boolean;
  crf: number;
  file_size_bytes: number | null;
  original_filename: string | null;
  duration_seconds: number | null;
  frames_total: number | null;
  frames_rendered: number;
  error_message: string | null;
  created_at: string;
  completed_at: string | null;
}

export const videoExportsApi = {
  create: (req: CreateVideoExportRequest) =>
    fetch('/api/video-exports', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(req),
    }).then(r => r.ok ? r.json() : r.json().then(e => Promise.reject(e))),

  list: (): Promise<VideoExportJob[]> =>
    fetch('/api/video-exports').then(r => r.json()),

  get: (id: string): Promise<VideoExportJob> =>
    fetch(`/api/video-exports/${id}`).then(r => r.json()),

  cancel: (id: string): Promise<void> =>
    fetch(`/api/video-exports/${id}`, { method: 'DELETE' }).then(() => undefined),

  getDownloadUrl: (id: string): string =>
    `/api/video-exports/${id}/download`,
};
```

## Gotchas

- **`graphicId` threading:** the playback bar needs to know what graphic is currently displayed. In Process, there should be a `selectedGraphicId` or similar piece of state already in scope — pass it down as a prop to the bar rather than pulling it from a store inside the bar (keeps the bar generic). Check how the process page currently identifies the active graphic.
- **Modal and `position: fixed`:** if the modal uses `position: fixed`, it works fine here (the playback bar is not inside react-grid-layout transforms). But verify the modal's backdrop covers the grid properly.
- **The Record button must NOT appear** in `TimeContextBar` mode (dashboards). The existing `barMode` dispatch already routes those to a separate component, so this is automatic — but confirm.
- **Locale for datetime inputs:** `<input type="datetime-local">` value must be in `YYYY-MM-DDTHH:mm` format. Convert epoch ms:
  ```typescript
  const toDatetimeLocal = (ms: number) => {
    const d = new Date(ms);
    const pad = (n: number) => String(n).padStart(2, '0');
    return `${d.getFullYear()}-${pad(d.getMonth()+1)}-${pad(d.getDate())}T${pad(d.getHours())}:${pad(d.getMinutes())}`;
  };
  ```
